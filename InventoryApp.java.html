import java.sql.*;

public class InventoryApp {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/InventoryDB?useSSL=false&allowPublicKeyRetrieval=true";
        String user = "root";
        String password = "jaan";  // Replace with your actual password

        try {
            Connection conn = DriverManager.getConnection(url, user, password);
            System.out.println("‚úÖ Database Connected Successfully!");

            // Create Statement
            Statement stmt = conn.createStatement();
            String query = "SELECT * FROM inventory";

            // Execute Query
            ResultSet rs = stmt.executeQuery(query);

            // Display Data
            System.out.println("üì¶ Inventory List:");
            while (rs.next()) {
                System.out.println("ID: " + rs.getInt("id") +
                        ", Name: " + rs.getString("item_name") +
                        ", Quantity: " + rs.getInt("quantity") +
                        ", Price: " + rs.getDouble("price") +
                        ", Category: " + rs.getString("category") +
                        ", Added On: " + rs.getTimestamp("created_at"));
            }

            // Close Connections
            rs.close();
            stmt.close();
            conn.close();
        } catch (SQLException e) {
            System.out.println("‚ùå Connection Failed!");
            e.printStackTrace();
        }
    }
}
